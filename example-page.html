Basic Inline Calendar

$('[data-toggle=calendar]').calendar();

Recurring unavailable dates

$('[data-toggle=calendar]').calendar({
          unavailable: ['*-*-8', '*-*-10']
          holidays: ['2017-11-25', '2017-11-26', '2017-11-27', '2017-11-28']
});

First Day = Tuesday

$('[data-toggle=calendar]').calendar({
          day_first: 2,
          unavailable: ['2017-11-10', '2017-11-25'],
});

Custom day names, month names (Vietnamese)

$('[data-toggle=calendar]').calendar({
          day_name: ['CN', 'Hai', 'Ba', 'Tư', 'Năm', 'Sáu', 'Bảy'],
          month_name: ['Tháng Một', 'Tháng Hai', 'Tháng Ba', 'Tháng Tư', 'Tháng Năm', 'Tháng Sáu', 'Tháng Bảy', 'Tháng Tám', 'Tháng Chín', 'Tháng Mười', 'Tháng Mười Một', 'Tháng Mười Hai'],
          unavailable: ['2017-11-10']
});

3 months with recurring unavailable dates

$('[data-toggle=calendar]').calendar({
          num_next_month: 3,
          num_prev_month: 3,
          unavailable: ['*-*-8', '*-*-10']
});

Entire year

$('[data-toggle=calendar]').calendar({
          year: 2017,
          month: 1
          num_next_month: 11,
});
